chatbot.pyChat History
Multi-turn conversational chatbot that maintains chat history using SystemMessage, HumanMessage, and AIMessage — so the model can remember past conversation context.
from langchain_google_genai import ChatGoogleGenerativeAI
from langchain_core.messages import SystemMessage, HumanMessage, AIMessage
from dotenv import load_dotenv

load_dotenv()

model = ChatGoogleGenerativeAI(model="gemini-2.5-flash")

chat_history = [
    SystemMessage(content="You are a helpful AI assistant.")
]

while True:
    user_input = input("You: ")
    if user_input.lower() == "exit":
        break
    chat_history.append(HumanMessage(content=user_input))
    result = model.invoke(chat_history)
    chat_history.append(AIMessage(content=result.content))
    print(f"AI: {result.content}")

print(chat_history)

structured_output_parser.pyStructuredOutputParser
Uses ResponseSchema + StructuredOutputParser to ask the model for three specific facts in JSON format. OutputFixingParser auto-corrects malformed output.

from dotenv import load_dotenv
from langchain_core.prompts import PromptTemplate
from langchain.output_parsers import StructuredOutputParser, ResponseSchema
from langchain.output_parsers import OutputFixingParser

model = ChatGoogleGenerativeAI(model="gemini-2.5-flash")

# Define schema
schema = [
    ResponseSchema(name="fact_1", description="First fact about black hole"),
    ResponseSchema(name="fact_2", description="Second fact about black hole"),
    ResponseSchema(name="fact_3", description="Third fact about black hole"),
]

parser = StructuredOutputParser.from_response_schemas(schema)
fixing_parser = OutputFixingParser.from_llm(llm=model, parser=parser)

template = PromptTemplate(
    template="Give me three facts about {topic}.\nReturn only valid JSON that follows this format:\n{response_format}",
    input_variables=["topic"],
    partial_variables={"response_format": parser.get_format_instructions()}
)

chain = template | model | fixing_parser

result = chain.invoke({"topic": "black hole"})
print(result)

conditional_chain.pyConditional Chain
Combines a classifier chain with a branch chain. First classifies feedback as positive or negative, then routes to an appropriate response prompt using RunnableBranch and RunnableLambda.
from langchain_google_genai import ChatGoogleGenerativeAI
from dotenv import load_dotenv
from langchain_core.prompts import PromptTemplate
from langchain_core.output_parsers import StrOutputParser, PydanticOutputParser
from langchain_core.runnables import RunnableBranch, RunnableLambda
from pydantic import BaseModel, Field
from typing import Literal

load_dotenv()

model = ChatGoogleGenerativeAI(model="gemini-2.5-flash")
parser = StrOutputParser()

class Feedback(BaseModel):
    sentiment: Literal["positive", "negative"] = Field(description="The sentiment of the feedback provided")

parser2 = PydanticOutputParser(pydantic_object=Feedback)

prompt1 = PromptTemplate(
    template="Classify the sentiment of the following feedback text into positive or negative:\n{feedback}\n{response_format}",
    input_variables=["feedback"],
    partial_variables={"response_format": parser2.get_format_instructions()}
)

prompt2 = PromptTemplate(
    template="Write an appropriate response to this positive feedback:\n{feedback}",
    input_variables=["feedback"]
)

prompt3 = PromptTemplate(
    template="Write an appropriate response to this negative feedback:\n{feedback}",
    input_variables=["feedback"]
)

# Classifier chain
classifier_chain = prompt1 | model | parser2

# Branch chain
branch_chain = RunnableBranch(
    (lambda x: x.sentiment == "positive", prompt2 | model | parser),
    (lambda x: x.sentiment == "negative", prompt3 | model | parser),
    RunnableLambda(lambda x: "No valid sentiment found")
)

# Combined chain
chain = classifier_chain | branch_chain

result = chain.invoke({"feedback": "The phone is actually amazing!"})
print(result)
parallel_chain.pyParallel Chain
Runs two chains in parallel — one generates notes, another generates quiz questions from the same topic. Both outputs are then merged into a single document by a final chain.
from langchain_google_genai import ChatGoogleGenerativeAI
from dotenv import load_dotenv
from langchain_core.prompts import PromptTemplate
from langchain_core.output_parsers import StrOutputParser
from langchain_core.runnables import RunnableParallel

load_dotenv()

model = ChatGoogleGenerativeAI(model="gemini-2.5-flash")
parser = StrOutputParser()

prompt1 = PromptTemplate(
    template="Generate short and simple notes for the following topic:\n{topic}",
    input_variables=["topic"]
)

prompt2 = PromptTemplate(
    template="Generate five short question-answers from the following text:\n{topic}",
    input_variables=["topic"]
)

prompt3 = PromptTemplate(
    template="Merge the provided notes and quiz into a single document.\nNotes: {notes}\nQuiz: {quiz}",
    input_variables=["notes", "quiz"]
)

# Parallel chain runs notes and quiz simultaneously
runnable_chain = RunnableParallel({
    "notes": prompt1 | model | parser,
    "quiz":  prompt2 | model | parser
})

# Final chain merges both outputs
final_chain = prompt3 | model | parser

chain = runnable_chain | final_chain

text = """Support Vector Machines (SVM) are supervised learning models used for
classification and regression. They find the optimal hyperplane that maximizes
the margin between different classes."""

result = chain.invoke({"topic": text})
print(result)

basic_rag.pyRAG Pipeline
Full RAG pipeline: loads a text file → splits into chunks → embeds with Google Generative AI embeddings → stores in FAISS vector store → creates retriever → answers questions from the document.
from langchain_google_genai import ChatGoogleGenerativeAI, GoogleGenerativeAIEmbeddings
from dotenv import load_dotenv
from langchain_community.document_loaders import TextLoader
from langchain.text_splitter import RecursiveCharacterTextSplitter
from langchain_community.vectorstores import FAISS
from langchain.chains import RetrievalQA

load_dotenv()

# Step 1: Load document
loader = TextLoader("docs.txt")
documents = loader.load()

# Step 2: Split text into smaller chunks
text_splitter = RecursiveCharacterTextSplitter(
    chunk_size=500,
    chunk_overlap=50
)
docs = text_splitter.split_documents(documents)

# Step 3: Create embeddings
embeddings = GoogleGenerativeAIEmbeddings(model="models/gemini-embedding-001")

# Step 4: Store embeddings in FAISS vector store
vector_store = FAISS.from_documents(docs, embeddings)

# Step 5: Create a retriever
retriever = vector_store.as_retriever()

# Step 6: Initialize the LLM
llm = ChatGoogleGenerativeAI(model="gemini-2.5-flash")

# Step 7: Create a RetrievalQA chain
chain = RetrievalQA.from_chain_type(llm=llm, retriever=retriever)

# Step 8: Query
query = "What are the key takeaways from the document?"
response = chain.invoke(query)
print(response["result"])

Graph with conditional routing. A router node reads the operation field and directs flow to either the adder node or the subtractor node using add_conditional_edges().
from typing import TypedDict
from langgraph.graph import StateGraph, START, END

class AgentState(TypedDict):
    number1: int
    number2: int
    operation: str
    result: int

def adder(state: AgentState) -> AgentState:
    """This node adds two numbers."""
    state["result"] = state["number1"] + state["number2"]
    return state

def subtract(state: AgentState) -> AgentState:
    """This node subtracts two numbers."""
    state["result"] = state["number1"] - state["number2"]
    return state

def decide_next_node(state: AgentState) -> str:
    """Router: selects the next node based on operation."""
    if state["operation"] == "+":
        return "add_op"
    elif state["operation"] == "-":
        return "sub_op"

graph.add_conditional_edges("router", decide_next_node, {
    "add_op": "add_node",
    "sub_op": "sub_node"

graph.add_edge("add_node", END)
graph.add_edge("sub_node", END)

    results = []
    for t in tool_calls:
        t_name = t["name"]
        t_id   = t["id"]
        t_args = t["args"]
        print(f"Executing tool: {t_name} with input: {t_args}")
        if t_name in tool_map:
            result = tool_map[t_name].invoke(t_args.get("query", ""))
        else:
            result = "Tool not found"
        print(f"Tool result: {result[:100]}...")
        results.append(ToolMessage(content=str(result), tool_call_id=t_id, name=t_name))
    print("Tools executed.")
    return {"messages": results}

# ── 8. Build graph ─────────────────────────────────────────────────────────
graph = StateGraph(AgentState)
graph.add_node("llm",             call_llm)
graph.add_node("retriever_agent", take_action)

graph.add_edge(START, "llm")
graph.add_conditional_edges("llm", should_continue, {True: "retriever_agent", False: END})
graph.add_edge("retriever_agent", "llm")

rag_agent = graph.compile()

# ── 9. Run agent ───────────────────────────────────────────────────────────
def run_agent():
    while True:
        user_input = input("What is your question? (type 'exit' to quit): ")
        if user_input.lower() in ["exit", "quit"]:
            break
        messages = [HumanMessage(content=user_input)]
        response = rag_agent.invoke({"messages": messages})
        print("\nAgent Response:", response["messages"][-1].content)

run_agent()
